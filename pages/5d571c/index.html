# 发送网络图片

const { segment } = require('oicq');

// 目标图片链接，支持 "http(s)://"，"base:64//"和本地文件和收到的file
let targetUrl = 'https://nilbridge.site/img/index.jpg';

let pic = segment.image(targetUrl);

// 是否作为自定义表情发送
pic.asface = true;

//然后把这个pic对象send出去就可以
